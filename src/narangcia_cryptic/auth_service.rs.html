1//! # Authentication Service Module
2//!
3//! This module provides the high-level logic for user authentication, acting as the central point of the crate's interactions.
4//! It exposes the [`AuthService`] struct, which aggregates all necessary dependencies and provides methods for user registration,
5//! login, token management, and user retrieval. The service is designed to be flexible and extensible, allowing for custom
6//! password managers, user repositories, and token services.
7//!
8//! ## Features
9//! - User registration and login
10//! - Password verification and management
11//! - Token generation, validation, and refresh
12//! - User retrieval from tokens
13//! - Extensible via dependency injection
14
15use std::sync::Arc;

17use crate::{core::user::User, error::AuthError};

19/// The main structure of the authentication service.
20/// It aggregates the necessary dependencies to perform operations.
21/// The main structure of the authentication service.
22///
23/// `AuthService` aggregates the necessary dependencies to perform authentication operations, such as user registration,
24/// login, token management, and user retrieval. It is designed to be flexible and allows for custom implementations of
25/// password managers, user repositories, and token services.
26pub struct AuthService {
27    /// Shared configuration and variables for the authentication service.
28    pub vars: Arc<crate::core::vars::AuthServiceVariables>,
29    /// The password manager responsible for password hashing and verification.
30    pub password_manager: Box<dyn crate::core::password::SecurePasswordManager + Send + Sync>,
31    /// The user repository for persistent user management (e.g., database or in-memory store).
32    pub persistent_users_manager:
33        Box<dyn crate::core::user::persistence::UserRepository + Send + Sync>,
34    /// The token manager responsible for generating and validating authentication tokens.
35    pub token_manager: Box<dyn crate::core::token::TokenService + Send + Sync>,
36}
37
38impl Default for AuthService {
39    /// Creates a default [`AuthService`] instance using default variables, Argon2 password manager,
40    /// in-memory user repository, and JWT token service.
41    fn default() -> Self {
42        let vars = Arc::new(crate::core::vars::AuthServiceVariables::default());
43        Self {
44            vars: vars.clone(),
45            password_manager: Box::new(crate::core::password::Argon2PasswordManager::default()),
46            persistent_users_manager: Box::new(
47                crate::core::user::persistence::InMemoryUserRepo::new(),
48            ),
49            token_manager: Box::new(crate::core::token::jwt::JwtTokenService::new(
50                &vars.secret_key,
51                vars.token_expiration,
52                vars.refresh_token_expiration,
53            )),
54        }
55    }
56}
57
58impl AuthService {
59    /// Constructs a new [`AuthService`] with the provided dependencies.
60    ///
61    /// # Arguments
62    /// * `vars` - Shared configuration and variables for the authentication service.
63    /// * `password_manager` - Optional custom password manager. If `None`, uses Argon2 by default.
64    /// * `persistent_users_manager` - Optional custom user repository. If `None`, uses in-memory repository by default.
65    /// * `token_manager` - Optional custom token service. If `None`, uses JWT token service by default.
66    ///
67    /// # Returns
68    /// Returns an [`AuthService`] instance or an [`AuthError`] if construction fails.
69    pub fn new(
70        vars: Arc<crate::core::vars::AuthServiceVariables>,
71        password_manager: Option<
72            Box<dyn crate::core::password::SecurePasswordManager + Send + Sync>,
73        >,
74        persistent_users_manager: Option<
75            Box<dyn crate::core::user::persistence::UserRepository + Send + Sync>,
76        >,
77        token_manager: Option<Box<dyn crate::core::token::TokenService + Send + Sync>>,
78    ) -> Result<Self, AuthError> {
79        let pwd_manager = match password_manager {
80            Some(manager) => manager,
81            None => Box::new(crate::core::password::Argon2PasswordManager::default()),
82        };
83        let pum = match persistent_users_manager {
84            Some(manager) => manager,
85            None => Box::new(crate::core::user::persistence::InMemoryUserRepo::new()),
86        };
87        let tk_manager = match token_manager {
88            Some(manager) => manager,
89            None => Box::new(crate::core::token::jwt::JwtTokenService::new(
90                &vars.secret_key,
91                vars.token_expiration,
92                vars.refresh_token_expiration,
93            )),
94        };
95
96        Ok(AuthService {
97            vars,
98            password_manager: pwd_manager,
99            persistent_users_manager: pum,
100            token_manager: tk_manager,
101        })
102    }
103
104    /// Registers a new user in the system.
105    ///
106    /// # Arguments
107    /// * `user` - The [`User`] object to register.
108    ///
109    /// # Returns
110    /// Returns `Ok(())` if registration is successful, or an [`AuthError`] if registration fails.
111    pub async fn signup(&self, user: User) -> Result<(), AuthError> {
112        self.persistent_users_manager
113            .add_user(user)
114            .await
115            .map(|_user| ()) // Discard the returned User, return ()
116            .map_err(|e| AuthError::NotImplemented(format!("signup: {e}")))
117    }
118
119    /// Attempts to log in a user by verifying their credentials.
120    ///
121    /// # Arguments
122    /// * `identifier` - The user identifier (e.g., username or email).
123    /// * `plain_password` - The user's plain text password.
124    ///
125    /// # Returns
126    /// Returns the [`User`] if login is successful, or an [`AuthError`] if authentication fails.
127    pub async fn login_with_credentials(
128        &self,
129        identifier: &str,
130        plain_password: &str,
131    ) -> Result<User, AuthError> {
132        // Find user by identifier (assuming identifier is stored in credentials.identifier)
133        let stored_user = self
134            .persistent_users_manager
135            .get_user_by_identifier(identifier)
136            .await
137            .ok_or(AuthError::InvalidCredentials)?;
138
139        // Verify the password using the password manager from the service
140        let is_valid = self
141            .password_manager
142            .verify_password(plain_password, &stored_user.credentials.password_hash)
143            .await
144            .map_err(|e| {
145                AuthError::PasswordVerificationError(format!("Password verification failed: {e}"))
146            })?;
147
148        if is_valid {
149            Ok(stored_user)
150        } else {
151            Err(AuthError::InvalidCredentials)
152        }
153    }
154
155    /// Generates a new token pair (access and refresh tokens) for a given user ID.
156    ///
157    /// # Arguments
158    /// * `id` - The user ID for which to generate tokens.
159    ///
160    /// # Returns
161    /// Returns a [`TokenPair`] containing access and refresh tokens, or an [`AuthError`] if generation fails.
162    pub async fn get_tokens(&self, id: String) -> Result<crate::core::token::TokenPair, AuthError> {
163        self.token_manager.generate_token_pair(&id).await
164    }
165
166    /// Validates an access token and returns the associated claims.
167    ///
168    /// # Arguments
169    /// * `token` - The access token to validate.
170    ///
171    /// # Returns
172    /// Returns the token claims if valid, or an [`AuthError`] if validation fails.
173    pub async fn validate_access_token(
174        &self,
175        token: &str,
176    ) -> Result<Box<dyn crate::core::token::claims::Claims + Send + Sync>, AuthError> {
177        self.token_manager.validate_access_token(token).await
178    }
179
180    /// Refreshes an access token using a valid refresh token.
181    ///
182    /// # Arguments
183    /// * `refresh_token` - The refresh token to use for generating a new access token.
184    ///
185    /// # Returns
186    /// Returns a new [`TokenPair`] if the refresh token is valid, or an [`AuthError`] if refresh fails.
187    pub async fn refresh_access_token(
188        &self,
189        refresh_token: &str,
190    ) -> Result<crate::core::token::TokenPair, AuthError> {
191        self.token_manager.refresh_access_token(refresh_token).await
192    }
193
194    /// Validates a token and extracts the user ID (subject) from it.
195    ///
196    /// # Arguments
197    /// * `token` - The token to validate and extract the user ID from.
198    ///
199    /// # Returns
200    /// Returns the user ID as a `String` if the token is valid, or an [`AuthError`] if validation fails.
201    pub async fn get_user_id_from_token(&self, token: &str) -> Result<String, AuthError> {
202        let claims = self.validate_access_token(token).await?;
203        Ok(claims.get_subject().to_string())
204    }
205
206    /// Checks if a token is expired by attempting to validate it.
207    ///
208    /// # Arguments
209    /// * `token` - The token to check for expiration.
210    ///
211    /// # Returns
212    /// Returns `true` if the token is expired or invalid, `false` otherwise.
213    pub async fn is_token_expired(&self, token: &str) -> bool {
214        self.validate_access_token(token).await.is_err()
215    }
216
217    /// Complete login flow that returns both the user and a new token pair.
218    ///
219    /// # Arguments
220    /// * `identifier` - The user identifier (e.g., username or email).
221    /// * `plain_password` - The user's plain text password.
222    ///
223    /// # Returns
224    /// Returns a tuple of the [`User`] and a [`TokenPair`] if login is successful, or an [`AuthError`] if authentication fails.
225    pub async fn login_with_credentials_and_tokens(
226        &self,
227        identifier: &str,
228        plain_password: &str,
229    ) -> Result<(User, crate::core::token::TokenPair), AuthError> {
230        let user = self
231            .login_with_credentials(identifier, plain_password)
232            .await?;
233        let tokens = self.get_tokens(user.id.clone()).await?;
234        Ok((user, tokens))
235    }
236
237    /// Validates a token and retrieves the associated user from the repository.
238    ///
239    /// # Arguments
240    /// * `token` - The token to validate and extract the user from.
241    ///
242    /// # Returns
243    /// Returns the [`User`] if the token is valid and the user exists, or an [`AuthError`] otherwise.
244    pub async fn get_user_from_token(&self, token: &str) -> Result<User, AuthError> {
245        let user_id = self.get_user_id_from_token(token).await?;
246        self.persistent_users_manager
247            .get_user_by_id(&user_id)
248            .await
249            .ok_or(AuthError::UserNotFound)
250    }
251}
