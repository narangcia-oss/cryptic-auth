1//! Password management trait for secure password hashing and verification.
2//!
3//! This module defines the [`SecurePasswordManager`] trait, which provides an abstraction
4//! for securely hashing and verifying passwords. Implementations of this trait are responsible
5//! for using cryptographically secure algorithms to protect user credentials.
6//!
7//! # Example
8//!
9//! ```rust,ignore
10//! use your_crate::core::password::manager::SecurePasswordManager;
11//!
12//! struct MyPasswordManager;
13//!
14//! #[async_trait::async_trait]
15//! impl SecurePasswordManager for MyPasswordManager {
16//!     async fn hash_password(&self, password: &str) -> Result<String, AuthError> {
17//!         // Implementation here
18//!         Ok("hashed".to_string())
19//!     }
20//!
21//!     async fn verify_password(&self, password: &str, hashed_password: &str) -> Result<bool, AuthError> {
22//!         // Implementation here
23//!         Ok(true)
24//!     }
25//! }
26//! ```
27
28use crate::error::AuthError;
29
30#[async_trait::async_trait]
31/// Trait for secure password management, including hashing and verification.
32///
33/// Implement this trait to provide password hashing and verification using a secure algorithm
34/// (e.g., Argon2, bcrypt, scrypt). The trait is asynchronous to support non-blocking operations
35/// in environments where password hashing may be computationally expensive.
36#[async_trait::async_trait]
37pub trait SecurePasswordManager {
38    /// Hashes a plaintext password using a secure algorithm.
39    ///
40    /// # Arguments
41    ///
42    /// * `password` - The plaintext password to hash.
43    ///
44    /// # Returns
45    ///
46    /// * `Ok(String)` containing the hashed password if successful.
47    /// * `Err(AuthError)` if hashing fails.
48    ///
49    /// # Example
50    ///
51    /// ```rust,ignore
52    /// let hashed = manager.hash_password("my_password").await?;
53    /// ```
54    async fn hash_password(&self, password: &str) -> Result<String, AuthError>;
55
56    /// Verifies a plaintext password against a hashed password.
57    ///
58    /// # Arguments
59    ///
60    /// * `password` - The plaintext password to verify.
61    /// * `hashed_password` - The hashed password to compare against.
62    ///
63    /// # Returns
64    ///
65    /// * `Ok(true)` if the password matches the hash.
66    /// * `Ok(false)` if the password does not match.
67    /// * `Err(AuthError)` if verification fails.
68    ///
69    /// # Example
70    ///
71    /// ```rust,ignore
72    /// let is_valid = manager.verify_password("my_password", &hashed).await?;
73    /// ```
74    async fn verify_password(
75        &self,
76        password: &str,
77        hashed_password: &str,
78    ) -> Result<bool, AuthError>;
79}
