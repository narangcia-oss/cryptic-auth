1//! OAuth2 provider, token, user info, config, and session types for authentication flows.
2//!
3//! This module defines enums and structs for representing OAuth2 providers, tokens, user information,
4//! configuration, and session state. It also provides utility methods for working with these types.
5//!
6//! # OAuth2 Flow with Frontend Redirect
7//!
8//! The OAuth2 implementation supports automatic redirection to frontend applications after successful
9//! authentication. Here's how the flow works:
10//!
11//! 1. **Generate Auth URL**: Client requests authorization URL from `/oauth/{provider}/auth`
12//! 2. **User Authorization**: User is redirected to provider (Google, GitHub, etc.) for authorization
13//! 3. **Provider Callback**: Provider redirects back to `/oauth/{provider}/callback` with auth code
14//! 4. **Token Exchange**: Server exchanges code for tokens automatically
15//! 5. **Frontend Redirect**: Server redirects user to `redirect_frontend_uri` with tokens in URL fragment
16//!
17//! ## Example Configuration
18//!
19//! ```rust
20//! use cryptic::core::oauth::store::OAuth2Config;
21//!
22//! let config = OAuth2Config {
23//!     app_name: "My App".to_string(),
24//!     client_id: "your-client-id".to_string(),
25//!     client_secret: "your-client-secret".to_string(),
26//!     redirect_uri: "https://api.myapp.com/oauth/google/callback".to_string(),
27//!     redirect_frontend_uri: "https://myapp.com/auth/callback".to_string(),
28//!     additional_scopes: vec!["profile".to_string()],
29//! };
30//! ```
31//!
32//! ## Frontend Token Extraction
33//!
34//! Your frontend application should handle the redirect and extract tokens from the URL fragment:
35//!
36//! ```javascript
37//! // At your redirect_frontend_uri (e.g., https://myapp.com/auth/callback)
38//! function handleOAuthCallback() {
39//!   const fragment = window.location.hash.substring(1);
40//!   const params = new URLSearchParams(fragment);
41//!
42//!   if (params.has('error')) {
43//!     console.error('OAuth error:', params.get('error'));
44//!   } else {
45//!     const accessToken = params.get('access_token');
46//!     const refreshToken = params.get('refresh_token');
47//!     const userId = params.get('user_id');
48//!
49//!     // Store tokens and proceed
50//!     localStorage.setItem('accessToken', accessToken);
51//!     // ... redirect to your app
52//!   }
53//! }
54//! ```

59/// Supported OAuth2 providers for authentication.
60///
61/// This enum lists all external providers supported by the authentication system.
62/// Each variant represents a third-party OAuth2 provider that can be used for user authentication.
63#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq, Hash)]
64#[serde(rename_all = "lowercase")]
65pub enum OAuth2Provider {
66    /// Google OAuth2 provider.
67    Google,
68    /// GitHub OAuth2 provider.
69    GitHub,
70    /// Discord OAuth2 provider.
71    Discord,
72    /// Microsoft OAuth2 provider.
73    Microsoft,
74}
75
76impl OAuth2Provider {
77    /// Returns the display name of the provider as a human-readable string.
78    ///
79    /// # Examples
80    ///
81    /// ```rust
82    /// let provider = OAuth2Provider::Google;
83    /// assert_eq!(provider.display_name(), "Google");
84    /// ```
85    pub fn display_name(&self) -> &'static str {
86        match self {
87            Self::Google => "Google",
88            Self::GitHub => "GitHub",
89            Self::Discord => "Discord",
90            Self::Microsoft => "Microsoft",
91        }
92    }
93
94    /// Returns the default OAuth2 scopes required for the provider.
95    ///
96    /// # Examples
97    ///
98    /// ```rust
99    /// let scopes = OAuth2Provider::Google.default_scopes();
100    /// assert!(scopes.contains(&"email"));
101    /// ```
102    pub fn default_scopes(&self) -> Vec<&'static str> {
103        match self {
104            Self::Google => vec!["openid", "email", "profile"],
105            Self::GitHub => vec!["user:email"],
106            Self::Discord => vec!["identify", "email"],
107            Self::Microsoft => vec!["openid", "email", "profile"],
108        }
109    }
110}

112/// Represents an OAuth2 token, including access and refresh tokens, expiration, and provider info.
113///
114/// This struct holds all relevant information about an OAuth2 token issued by a provider,
115/// including the access token, optional refresh token, expiration, token type, scope, provider, and creation time.
116#[derive(Debug, Clone, Serialize, Deserialize)]
117pub struct OAuth2Token {
118    /// The access token string used for authenticated requests.
119    pub access_token: String,
120    /// The optional refresh token string, used to obtain new access tokens.
121    pub refresh_token: Option<String>,
122    /// The expiration time of the access token, if available.
123    pub expires_at: Option<NaiveDateTime>,
124    /// The type of token (usually "Bearer").
125    pub token_type: String,
126    /// The scope of the token, if provided by the provider.
127    pub scope: Option<String>,
128    /// The OAuth2 provider that issued the token.
129    pub provider: OAuth2Provider,
130    /// The time the token was created.
131    pub created_at: NaiveDateTime,
132}
133
134impl OAuth2Token {
135    /// Checks if the token is expired.
136    ///
137    /// Returns `true` if the current time is past the expiration time, or `false` otherwise.
138    /// If `expires_at` is `None`, returns `false`.
139    ///
140    /// # Examples
141    ///
142    /// ```rust
143    /// let token = OAuth2Token { /* ... */ };
144    /// let expired = token.is_expired();
145    /// ```
146    pub fn is_expired(&self) -> bool {
147        self.expires_at
148            .map(|exp| chrono::Utc::now().naive_utc() > exp)
149            .unwrap_or(false)
150    }
151
152    /// Checks if the token will expire within the given threshold (in seconds).
153    ///
154    /// # Arguments
155    ///
156    /// * `threshold_secs` - Number of seconds to check for imminent expiration.
157    ///
158    /// Returns `true` if the token will expire within `threshold_secs` seconds, or `false` otherwise.
159    /// If `expires_at` is `None`, returns `false`.
160    ///
161    /// # Examples
162    ///
163    /// ```rust
164    /// let token = OAuth2Token { /* ... */ };
165    /// let soon = token.expires_soon(60);
166    /// ```
167    pub fn expires_soon(&self, threshold_secs: u64) -> bool {
168        self.expires_at
169            .map(|exp| {
170                let now = chrono::Utc::now().naive_utc();
171                let duration = exp.signed_duration_since(now).num_seconds();
172                duration < threshold_secs as i64
173            })
174            .unwrap_or(false)
175    }
176}

178/// Represents user information returned by an OAuth2 provider.
179///

198    pub locale: Option<String>,
199    /// The time the user info was last updated.
200    pub updated_at: NaiveDateTime,
201    /// The raw user info data as returned by the provider, if available.

203    pub raw_data: Option<serde_json::Value>,

206/// Configuration for OAuth2 authentication with a provider.
207///
208/// This struct defines the configuration required to set up OAuth2 authentication for a specific provider.
209/// It includes client credentials, redirect URIs, and additional scopes.
210#[derive(Debug, Clone)]
211pub struct OAuth2Config {
212    /// The name of the application using OAuth2.
213    pub app_name: String,
214    /// The OAuth2 client ID issued by the provider.
215    pub client_id: String,
216    /// The OAuth2 client secret issued by the provider.
217    pub client_secret: String,
218    /// The redirect URI for OAuth2 callbacks.
219    /// This is where the OAuth2 provider will redirect users after authorization.
220    /// It should point to your server's callback endpoint (e.g., `/oauth/{provider}/callback`).
221    pub redirect_callback_uri: String,
222    /// The redirect URI for the user to go back to the frontend application.
223    /// After successful OAuth2 authentication, the server will redirect the user to this URI
224    /// with authentication tokens included in the URL fragment (e.g., `#access_token=...&refresh_token=...`).
225    /// This should point to a frontend page that can handle token extraction from the URL fragment.
226    pub redirect_frontend_uri: String,
227    /// Additional scopes to request during authentication.
228    pub additional_scopes: Vec<String>,
229}

231impl OAuth2Config {
232    /// Returns the authorization URL for the given provider.
233    ///
234    /// # Arguments
235    ///
236    /// * `provider` - The OAuth2 provider for which to get the authorization URL.
237    ///
238    /// # Returns
239    ///
240    /// A static string containing the provider's authorization endpoint URL.
241    ///
242    /// # Examples
243    ///
244    /// ```rust
245    /// let config = OAuth2Config { /* ... */ };
246    /// let url = config.auth_url(OAuth2Provider::Google);
247    /// ```
248    pub fn auth_url(&self, provider: OAuth2Provider) -> &'static str {
249        match provider {
250            OAuth2Provider::Google => "https://accounts.google.com/o/oauth2/v2/auth",
251            OAuth2Provider::GitHub => "https://github.com/login/oauth/authorize",
252            OAuth2Provider::Discord => "https://discord.com/api/oauth2/authorize",
253            OAuth2Provider::Microsoft => {
254                "https://login.microsoftonline.com/common/oauth2/v2.0/authorize"
255            }
256        }
257    }
258
259    /// Returns the token endpoint URL for the given provider.
260    ///
261    /// # Arguments
262    ///
263    /// * `provider` - The OAuth2 provider for which to get the token endpoint URL.
264    ///
265    /// # Returns
266    ///
267    /// A static string containing the provider's token endpoint URL.
268    ///
269    /// # Examples
270    ///
271    /// ```rust
272    /// let config = OAuth2Config { /* ... */ };
273    /// let url = config.token_url(OAuth2Provider::Google);
274    /// ```
275    pub fn token_url(&self, provider: OAuth2Provider) -> &'static str {
276        match provider {
277            OAuth2Provider::Google => "https://oauth2.googleapis.com/token",
278            OAuth2Provider::GitHub => "https://github.com/login/oauth/access_token",
279            OAuth2Provider::Discord => "https://discord.com/api/oauth2/token",
280            OAuth2Provider::Microsoft => {
281                "https://login.microsoftonline.com/common/oauth2/v2.0/token"
282            }
283        }
284    }
285
286    /// Returns the user info endpoint URL for the given provider.
287    ///
288    /// # Arguments
289    ///
290    /// * `provider` - The OAuth2 provider for which to get the user info endpoint URL.
291    ///
292    /// # Returns
293    ///
294    /// A static string containing the provider's user info endpoint URL.
295    ///
296    /// # Examples
297    ///
298    /// ```rust
299    /// let config = OAuth2Config { /* ... */ };
300    /// let url = config.user_info_url(OAuth2Provider::Google);
301    /// ```
302    pub fn user_info_url(&self, provider: OAuth2Provider) -> &'static str {
303        match provider {
304            OAuth2Provider::Google => "https://www.googleapis.com/oauth2/v2/userinfo",
305            OAuth2Provider::GitHub => "https://api.github.com/user",
306            OAuth2Provider::Discord => "https://discord.com/api/users/@me",
307            OAuth2Provider::Microsoft => "https://graph.microsoft.com/v1.0/me",
308        }
309    }
310}
