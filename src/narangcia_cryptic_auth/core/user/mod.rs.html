1//! User data structures and persistence traits.
2//!
3//! This module provides the [`User`] struct, which encapsulates user identity and credentials,
4//! as well as methods for user creation with both hashed and plaintext passwords. It also
5//! re-exports the [`persistence`] submodule, which defines traits and types for user persistence
6//! operations (e.g., repositories).
7//!
8//! # Examples
9//!
10//! Creating a user with already hashed credentials:
11//! ```rust
12//! use cryptic::core::user::User;
13//! use cryptic::core::credentials::Credentials;
14//! let credentials = Credentials::default();
15//! let user = User::new("user-id".to_string(), credentials);
16//! ```
17//!
18//! Creating a user with a plaintext password (async):
19//! ```ignore
20//! use cryptic::core::user::User;
21//! use cryptic::core::credentials::PlainPassword;
22//! # async fn example(manager: &impl cryptic::core::password::SecurePasswordManager) {
23//! let user = User::with_plain_password(
24//!     manager,
25//!     "user-id".to_string(),
26//!     "username".to_string(),
27//!     PlainPassword::from("password123"),
28//! ).await.unwrap();
29//! # }
30//! ```
31
32use crate::core::credentials::{Credentials, PlainPassword};
33use crate::core::oauth::store::{OAuth2Provider, OAuth2UserInfo};
34use std::collections::HashMap;
35
36/// Represents a user in the authentication system.
37///
38/// The `User` struct contains a unique identifier and associated credentials.
39/// The credentials include the user's identifier (e.g., username or email) and password hash.
40#[derive(Debug, Clone)]
41pub struct User {
42    /// Unique identifier for the user (preferably a UUID).
43    pub id: String,
44    /// User credentials, including hashed password and identifier.
45    pub credentials: Option<Credentials>,
46    /// OAuth2 accounts linked to this user
47    pub oauth_accounts: HashMap<OAuth2Provider, OAuth2UserInfo>,
48    /// Account creation timestamp
49    pub created_at: chrono::NaiveDateTime,
50    /// Last updated timestamp
51    pub updated_at: chrono::NaiveDateTime,
52}
53
54impl Default for User {
55    fn default() -> Self {
56        let now = chrono::Utc::now().naive_utc();
57        Self {
58            id: String::new(),
59            credentials: None,
60            oauth_accounts: HashMap::new(),
61            created_at: now,
62            updated_at: now,
63        }
64    }
65}
66
67impl User {
68    /// Creates a new user with already hashed credentials.
69    ///
70    /// # Arguments
71    /// * `id` - Unique identifier for the user (e.g., UUID).
72    /// * `credentials` - User credentials, including hashed password and identifier.
73    ///
74    /// # Returns
75    /// A new [`User`] instance.
76    pub fn new(id: String, credentials: Credentials) -> Self {
77        let now = chrono::Utc::now().naive_utc();
78        Self {
79            id,
80            credentials: Some(credentials),
81            oauth_accounts: HashMap::new(),
82            created_at: now,
83            updated_at: now,
84        }
85    }
86
87    /// Creates a user from a plaintext password, hashing it using the provided password manager.
88    ///
89    /// # Arguments
90    /// * `manager` - Reference to a type implementing [`SecurePasswordManager`] for hashing passwords.
91    /// * `id` - Unique identifier for the user (e.g., UUID).
92    /// * `identifier` - User identifier (e.g., username or email).
93    /// * `plain_password` - The user's plaintext password.
94    ///
95    /// # Returns
96    /// * `Ok(User)` if the password was hashed and the user was created successfully.
97    /// * `Err(AuthError)` if password hashing or credential creation fails.
98    ///
99    /// # Errors
100    /// Returns [`AuthError`] if password hashing or credential creation fails.
101    ///
102    /// # Examples
103    /// ```ignore
104    /// # use cryptic::core::user::User;
105    /// # use cryptic::core::credentials::PlainPassword;
106    /// # async fn example(manager: &impl cryptic::core::password::SecurePasswordManager) {
107    /// let user = User::with_plain_password(
108    ///     manager,
109    ///     "user-id".to_string(),
110    ///     "username".to_string(),
111    ///     PlainPassword::from("password123"),
112    /// ).await.unwrap();
113    /// # }
114    /// ```
115    pub async fn with_plain_password(
116        manager: &(dyn crate::core::password::SecurePasswordManager + Send + Sync),
117        id: String,
118        identifier: String,
119        plain_password: PlainPassword,
120    ) -> Result<Self, crate::error::AuthError> {
121        let credentials =
122            Credentials::from_plain_password(manager, id.clone(), identifier, plain_password)
123                .await?;
124
125        Ok(Self {
126            id,
127            credentials: Some(credentials),
128            oauth_accounts: HashMap::new(),
129            created_at: chrono::Utc::now().naive_utc(),
130            updated_at: chrono::Utc::now().naive_utc(),
131        })
132    }
133
134    /// Links an OAuth account to this user.
135    ///
136    /// # Arguments
137    /// * `oauth_info` - The OAuth user info to link to this user.
138    ///
139    /// # Returns
140    /// The updated user with the OAuth account linked.
141    pub fn link_oauth_account(mut self, mut oauth_info: OAuth2UserInfo) -> Self {
142        oauth_info.user_id = self.id.clone();
143        self.oauth_accounts.insert(oauth_info.provider, oauth_info);
144        self.updated_at = chrono::Utc::now().naive_utc();
145        self
146    }
147
148    /// Unlinks an OAuth account from this user.
149    ///
150    /// # Arguments
151    /// * `provider` - The OAuth provider to unlink.
152    ///
153    /// # Returns
154    /// True if the account was unlinked, false if it wasn't linked.
155    pub fn unlink_oauth_account(&mut self, provider: OAuth2Provider) -> bool {
156        let was_linked = self.oauth_accounts.remove(&provider).is_some();
157        if was_linked {
158            self.updated_at = chrono::Utc::now().naive_utc();
159        }
160        was_linked
161    }
162
163    /// Checks if an OAuth account is linked to this user.
164    ///
165    /// # Arguments
166    /// * `provider` - The OAuth provider to check.
167    ///
168    /// # Returns
169    /// True if the provider is linked, false otherwise.
170    pub fn has_oauth_account(&self, provider: OAuth2Provider) -> bool {
171        self.oauth_accounts.contains_key(&provider)
172    }
173
174    /// Gets the OAuth account info for a specific provider.
175    ///
176    /// # Arguments
177    /// * `provider` - The OAuth provider to get info for.
178    ///
179    /// # Returns
180    /// The OAuth user info if linked, None otherwise.
181    pub fn get_oauth_account(&self, provider: OAuth2Provider) -> Option<&OAuth2UserInfo> {
182        self.oauth_accounts.get(&provider)
183    }
184
185    /// Creates a new user from OAuth account info only (no password credentials).
186    ///
187    /// # Arguments
188    /// * `id` - Unique identifier for the user (e.g., UUID).
189    /// * `oauth_info` - The OAuth user info to create the user with.
190    ///
191    /// # Returns
192    /// A new [`User`] instance with the OAuth account linked.
193    pub fn from_oauth(id: String, mut oauth_info: OAuth2UserInfo) -> Self {
194        oauth_info.user_id = id.clone();
195        let now = chrono::Utc::now().naive_utc();
196        let mut oauth_accounts = HashMap::new();
197        oauth_accounts.insert(oauth_info.provider, oauth_info);
198
199        Self {
200            id,
201            credentials: None,
202            oauth_accounts,
203            created_at: now,
204            updated_at: now,
205        }
206    }
207}
