1//! Password Management Module
2//!
3//! This module provides secure password hashing and verification functionality for authentication systems.
4//! It exposes password management strategies and utilities, including Argon2-based password hashing.
5//!
6//! # Modules
7//!
8//! - [`argon2`]: Contains the Argon2 password hashing implementation and configuration.
9//! - [`manager`]: Defines the `SecurePasswordManager` trait and related password management logic.
10//!
11//! # Re-exports
12//!
13//! - [`Argon2PasswordManager`]: A concrete password manager using Argon2 for hashing and verification.
14//! - [`SecurePasswordManager`]: The main trait for password management operations.
15//!
16//! # Example
17//!
18//! ```rust
19//! use cryptic::core::password::{Argon2PasswordManager, SecurePasswordManager};
20//!
21//! let manager = Argon2PasswordManager::default();
22//! let password = "mysecret";
23//! let hash = manager.hash_password(password)?;
24//! assert!(manager.verify_password(password, &hash)?);
25//! ```
26//!
27//! # Security
28//!
29//! Always use a secure, up-to-date password hashing algorithm such as Argon2. Never store plain-text passwords.
30//! This module is designed to make it easy to follow best practices for password security.
31
32pub mod argon2;
33pub mod manager;
34
35/// Re-export of the Argon2-based password manager implementation.
36pub use argon2::Argon2PasswordManager;
37
38/// Re-export of the main password management trait.
