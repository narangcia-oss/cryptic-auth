1//! Security policy module.
2//!
3//! This module defines security policies and rules for authentication,
4//! such as password complexity requirements. It provides the [`PasswordPolicy`] struct,
5//! which allows you to specify and validate password strength requirements.
6//!
7//! # Example
8//!
9//! ```rust
10//! use cryptic::core::policy::PasswordPolicy;
11//!
12//! let policy = PasswordPolicy::default();
13//! assert!(policy.validate_password("Str0ng!Passw0rd").is_ok());
14//! ```
15
16use crate::error::AuthError;
17
18/// Represents the requirements for a strong password.
19///
20/// This struct allows you to configure password complexity rules such as minimum length,
21/// and whether to require uppercase, lowercase, digit, and special characters.
22///
23/// # Fields
24/// - `min_length`: Minimum number of characters required.
25/// - `require_uppercase`: If `true`, at least one uppercase letter is required.
26/// - `require_lowercase`: If `true`, at least one lowercase letter is required.
27/// - `require_digit`: If `true`, at least one digit is required.
28/// - `require_special_char`: If `true`, at least one non-alphanumeric character is required.
29pub struct PasswordPolicy {
30    pub min_length: usize,
31    pub require_uppercase: bool,
32    pub require_lowercase: bool,
33    pub require_digit: bool,
34    pub require_special_char: bool,
35}
36
37impl Default for PasswordPolicy {
38    /// Returns a [`PasswordPolicy`] with strong default requirements:
39    ///
40    /// - Minimum length: 12
41    /// - Requires uppercase, lowercase, digit, and special character
42    fn default() -> Self {
43        PasswordPolicy {
44            min_length: 12,
45            require_uppercase: true,
46            require_lowercase: true,
47            require_digit: true,
48            require_special_char: true,
49        }
50    }
51}
52
53impl PasswordPolicy {
54    /// Validates whether a password meets the requirements defined by this policy.
55    ///
56    /// # Arguments
57    /// * `password` - The password string to validate.
58    ///
59    /// # Returns
60    /// * `Ok(())` if the password satisfies all requirements.
61    /// * `Err(AuthError)` with a descriptive message if any requirement is not met.
62    ///
63    /// # Example
64    /// ```rust
65    /// use cryptic::core::policy::PasswordPolicy;
66    /// let policy = PasswordPolicy::default();
67    /// assert!(policy.validate_password("Str0ng!Passw0rd").is_ok());
68    /// assert!(policy.validate_password("weak").is_err());
69    /// ```
70    pub fn validate_password(&self, password: &str) -> Result<(), AuthError> {
71        if password.len() < self.min_length {
72            return Err(AuthError::InvalidInput(format!(
73                "Password must be at least {} characters long.",
74                self.min_length
75            )));
76        }
77
78        if self.require_uppercase && !password.chars().any(|c| c.is_ascii_uppercase()) {
79            return Err(AuthError::InvalidInput(
80                "Password must contain at least one uppercase letter.".to_string(),
81            ));
82        }
83        if self.require_lowercase && !password.chars().any(|c| c.is_ascii_lowercase()) {
84            return Err(AuthError::InvalidInput(
85                "Password must contain at least one lowercase letter.".to_string(),
86            ));
87        }
88        if self.require_digit && !password.chars().any(|c| c.is_ascii_digit()) {
89            return Err(AuthError::InvalidInput(
90                "Password must contain at least one digit.".to_string(),
91            ));
92        }
93        if self.require_special_char && !password.chars().any(|c| !c.is_ascii_alphanumeric()) {
94            return Err(AuthError::InvalidInput(
95                "Password must contain at least one special character.".to_string(),
96            ));
97        }
98        Ok(())
99    }
